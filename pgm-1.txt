#! /bin/sh
echo "Enter a pid:"
read pid
min=1
max=32768
if [ $pid -ge $min -a $pid -le $max ]
	then
		if ps -p $pid > /dev/null 2>&1
			then 
			kill $pid
			echo "\ngiven pid is a valid one and the process is killed\n"
		else
		 	echo "given pid is valid but not in a running state\n"
		fi
	else
	
	echo "oops!!! given an invalid pid.."
fi
